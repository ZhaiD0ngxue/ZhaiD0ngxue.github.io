iconv -f gb2312 -t utf8 tessss.txt -o win.txt    #将gb2312字符集格式的tessss.txt文档转换为utf8格式,并命名为win.txt  

[root@wang data]# file tessss.txt 
tessss.txt: ISO-8859 text, with no line terminators
[root@wang data]# file win.txt 
win.txt: UTF-8 Unicode text, with no line terminators
[root@wang data]#


iconv -l		#显示支持字符集编码的列表

-f 原始的字符集
-t 要转换的字符集
-o 输出的文件名
