==自己写的,格式很丑,但也没有问题

==更专业,格式更整齐的写法,使用tab对齐
[root@c803-81 ~]# cat /var/named/zdx.org.zone 
$ORIGIN .
$TTL 86400      ; 1 day
zdx.org                 IN SOA  master.zdx.org. admin.zdx.org. (
                                3          ; serial
                                7200       ; refresh (2 hours)
                                600        ; retry (10 minutes)
                                604800     ; expire (1 week)
                                86400      ; minimum (1 day)
                                )
                        NS      master.zdx.org.
$ORIGIN zdx.org.
master                  A       10.0.0.81
www                     A       10.0.0.101
[root@c803-81 ~]# cat -A  /var/named/zdx.org.zone 
$ORIGIN .$
$TTL 86400^I; 1 day$
zdx.org^I^I^IIN SOA^Imaster.zdx.org. admin.zdx.org. ($
^I^I^I^I3          ; serial$
^I^I^I^I7200       ; refresh (2 hours)$
^I^I^I^I600        ; retry (10 minutes)$
^I^I^I^I604800     ; expire (1 week)$
^I^I^I^I86400      ; minimum (1 day)$
^I^I^I^I)$
^I^I^INS^Imaster.zdx.org.$
$ORIGIN zdx.org.$
master^I^I^IA^I10.0.0.81$
www^I^I^IA^I10.0.0.101$
[root@c803-81 ~]# 
==

DNS允许动态更新

服务器端配置

# 修改域配置信息,添加allow-update
[root@c803-81 ~]# vim /etc/named.rfc1912.zones 
zone "zdx.org" IN {
        type master;
        file "zdx.org.zone";
        allow-update {any;};		# 修改这行,指定谁可以update,写IP也可以
};

# 千万别忘记reload
[root@c803-81 ~]# rndc reload

# 给zone目录添加权限
[root@c803-81 ~]# chmod 770 /var/named/

> zone zdx.org

> send

DNS服务器配置

# step1,修改主配置文件
[root@c803-81 ~]# vim /etc/named.conf 
# 这两行不注释也可以,修改如下
# 注释的含义其实也相同,也相当于修改成了下面这两行
listen-on port 53 { localhost; };
allow-query     { any; };
		
[root@c803-81 ~]# named-checkconf 

# step2,修改区域配置
[root@c803-81 ~]# vim /etc/named.rfc1912.zones

zone "0.0.10.in-addr.arpa" IN {
        type master;
        file "10.0.0.zone";
};

[root@c803-81 ~]# named-checkconf 

# step3,创建解析文件
[root@c803-81 ~]# touch /var/named/10.0.0.zone
[root@c803-81 ~]# chmod 640 /var/named/10.0.0.zone

[root@c803-81 ~]# ll /var/named/10.0.0.zone

[root@c803-81 ~]# 


# 添加解析配置
[root@c803-81 ~]# vim /var/named/10.0.0.zone
$TTL 1D
0.0.10.in-addr.arpa.  IN  SOA  ns.zdx.org.  admin.zdx.org.  (
  1  ;
  1H  ;
  5M  ;
  7D  ;
  1D  ;
)
  IN  NS  ns.zdx.org.
91  IN PTR  www.zdx.org.
101  IN PTR  wap.zdx.org.
# 注意
# 这里只是反向解析IP的最后一位,
# 如果反向解析的是IP的后两位,比如要反向解析,10.0.*.*,应该要注意什么?
# 一个是上面的区域配置要对应修改为:
# zone "0.10.in-addr.arpa" IN {
# 另一个就是这里,比如要解析10.0.1.2,正确的写法应该是:
# 2.1  IN  PTR  ftp.zdx.org
# 也需要反着来写

DNS转发

  1. 场景,应用在父域和子域系统的子域DNS服务器上面,当子域的客户端访问互联网时,子域DNS应首先查询父域DNS服务器的缓存以提高效率,所以需要启用转发
  2. 根据规则可以分为为全局转发和特定域转发
  3. 根据行为可以分为first(先查询父DNS服务器,如果未能获取answer则自行去递归查询互联网根DNS服务器)和only(仅查询父DNS服务器,如果未能获取answer则返回失败).

# 安装服务
[root@c803-81 ~]# dnf -y install bind bind-utils
[root@c803-81 ~]# systemctl start named

# 修改主配置文件,添加ACL
[root@c803-81 ~]# vim /etc/named.conf 

# 定义了ACL
acl beijingnet {
        10.0.0.61;		# 可以写网段,如10.0.0.0/24
};

acl shanghainet {
        10.0.0.71;
};

acl othernet {
        any;
};
# 定义view,将ACL和解析文件对应起来
view beijingview {
        match-clients {beijingnet;};
        zone "zdx.org" {
                type master;
                file "zdx.org.zone.beijing";
        };
        include "/etc/named.rfc1912.zones";
};

view shanghaiview {
        match-clients {shanghainet;};
        zone "zdx.org" {
                type master;
                file "zdx.org.zone.shanghai";
        };
        include "/etc/named.rfc1912.zones";
};
view otherview {
        match-clients {othernet;};
        include "/etc/named.rfc1912.zones";
};

# 如果使用了view,所有的zone都应该被包含在view中,所以要把最外面的的这下面一行注释掉.
#include "/etc/named.rfc1912.zones";
# 要把下面一段也注释掉,转移到named.rfc1912.zones配置文件中去.
#zone "." IN {
#        type hint;
#        file "named.ca";
#};

[root@c803-81 ~]# vim /etc/named.rfc1912.zones 
# 转移添加如下一段内容
zone "." IN {
        type hint;
        file "named.ca";
};
